184 void Text::remake_texture() {
185  if (texture != nullptr) {
186  SDL_DestroyTexture(texture);
187  } else {
188  /* Nothing to do. */
189  }
190 
191  SDL_Surface* surface;
192  switch (style) {
193  case SOLID:
194  surface = TTF_RenderText_Solid(font.get(), text.c_str(), color);
195  break;
196  case SHADED:
197  surface = TTF_RenderText_Shaded(font.get(), text.c_str(), color,
198  BLACK_BACKGROUND_ON_SDL_RGBA_COLOR);
199  break;
200  case BLENDED:
201  surface = TTF_RenderText_Blended(font.get(), text.c_str(), color);
202  break;
203  }
204 
205  if (surface == nullptr) {
206  printf("%s\n", SDL_GetError());
207  exit(-3);
208  } else {
209  /* Nothing to do. */
210  }
211 
212  texture = SDL_CreateTextureFromSurface(Game::get_instance().get_renderer(),
213  surface);
214  SDL_FreeSurface(surface);
215 
216  int w, h;
217  int query_texture = SDL_QueryTexture(texture, nullptr, nullptr, &w, &h);
218  if (query_texture) {
219  printf("Remake texture: %s\n", SDL_GetError());
220  exit(-1);
221  } else {
222  /* Nothing to do. */
223  }
224 
225  box.set_width(w);
226  box.set_height(h);
227  clip_rect.w = w;
228  clip_rect.h = h;
229 }
