116 void OptionsState::update(float) {
117  process_input();
118 
119  InputManager *input_manager = InputManager::get_instance();
120  assert(input_manager != nullptr);
121 
122  // Inputs
123  if (not input_manager->quit_requested()) {
124  /* Nothing to do. */
125  } else {
126  m_quit_requested = true;
127  return;
128  }
129 
136  if (pressed[DOWN]) {
137  changed.play();
138 
139  if (not on_submenu) {
140  assert(current_option <= static_cast<int>(options.size() - 1));
141 
142  if (current_option != static_cast<int>(options.size()) - 1) {
143  current_option++;
144  } else {
145  /* Nothing to do. */
146  }
147  } else {
148  string text = options[current_option]->get_text();
149  assert(text.empty() != true);
150 
151  assert(current_sub_option[current_option] <=
152  static_cast<int>(sub_options[text].size()) - 1);
153 
154  if (current_sub_option[current_option] !=
155  static_cast<int>(sub_options[text].size()) - 1) {
156  current_sub_option[current_option]++;
157  } else {
158  /* Nothing to do. */
159  }
160  }
161  } else if (pressed[UP]) {
162  changed.play();
163 
164  if (not on_submenu) {
165  assert(current_option >= 0);
166  if (current_option != 0) {
167  current_option--;
168  } else {
169  /* Nothing to do. */
170  }
171  } else {
172  assert(current_sub_option[current_option] >= 0);
173  if (current_sub_option[current_option] != 0) {
174  current_sub_option[current_option]--;
175  } else {
176  /* Nothing to do. */
177  }
178  }
179  } else {
180  /* Nothing to do. */
181  }
182 
190  if (pressed[START] or pressed[A]) {
191  selected.play();
192 
193  assert(current_option >= 0);
194  assert(current_option <= static_cast<int>(options.size()) - 1);
195 
196  if (not on_submenu) {
197  if (current_option ==
198  static_cast<int>(options.size()) - 1) { // Back button
199  m_quit_requested = true;
200  Game::get_instance().push(new MenuState(true));
201  return;
202  } else if (current_option == 2) { // Joysticks test
203  m_quit_requested = true;
204  Game::get_instance().push(new JoystickConfigState(0));
205  return;
206  } else if (current_option == 3) { // Keyboard controls
207  m_quit_requested = true;
208  Game::get_instance().push(new JoystickConfigState(0, true));
209  return;
210  } else {
211  on_submenu = true;
212  current_sub_option[current_option] =
213  get_current_sub_option(current_option);
214  }
215  } else {
221  if (current_option == 0) {
222  vector<pair<int, int> > resolutions = {
223  ii(RESOLUTION_1_WIDTH, RESOLUTION_1_HEIGHT),
224  ii(RESOLUTION_2_WIDTH, RESOLUTION_2_HEIGHT),
225  ii(RESOLUTION_3_WIDTH, RESOLUTION_3_HEIGHT),
226  ii(RESOLUTION_4_WIDTH, RESOLUTION_4_HEIGHT),
227  ii(RESOLUTION_5_WIDTH, RESOLUTION_5_HEIGHT)};
228 
229  int id_resolution = current_sub_option[current_option];
230  int new_width = resolutions[id_resolution].first;
231  int new_height = resolutions[id_resolution].second;
232 
233  Game::get_instance().change_resolution(new_width, new_height);
234  } else if (current_option == 1) {
235  bool fullscreen =
236  (current_sub_option[current_option] == 0 ? false : true);
237  Game::get_instance().set_fullscreen(fullscreen);
238  } else {
239  /* Nothing to do. */
240  }
241  }
242  } else if (pressed[SELECT] or pressed[B]) {
243  if (on_submenu) {
244  // FIXME insert back sound
245  selected.play();
246  on_submenu = false;
247 
248  for (unsigned i = 0; i < options.size(); i++) {
249  int id_option = get_current_sub_option(i);
250  assert(id_option >= 0);
251 
252  current_sub_option[i] = id_option;
253  }
254  } else {
255  selected.play();
256  m_quit_requested = true;
257  Game::get_instance().push(new MenuState(true));
258  return;
259  }
260  } else {
261  /* Nothing to do. */
262  }
263 
268  for (int i = 0; i < static_cast<int>(options.size()); i++) {
269  Text *cur_text = options[i];
270 
271  int prev_text_size = 1;
272 
273  if (i) {
274  prev_text_size =
275  std::max(static_cast<int>(
276  sub_options[options[i - 1]->get_text()].size()),
277  1);
278  } else {
279  /* Nothing to do. */
280  }
281  int prev_text_height = (TEXT_HEIGHT + TEXT_OFFSET * 2) * prev_text_size;
282 
283  int text_position_axi_x = 250;
284  int text_position_axi_y =
285  (i ? options[i - 1]->get_y() + prev_text_height : 200);
286 
287  cur_text->set_pos(text_position_axi_x, text_position_axi_y,
288  NOT_CENTER_HORIZONTALLY, NOT_CENTER_VERTICALLY);
289 
290  for (int j = 0;
291  j < static_cast<int>(sub_options[cur_text->get_text()].size());
292  j++) {
293  text_position_axi_x = 800;
294 
295  if (j) {
296  Text *prev_option = sub_options[cur_text->get_text()][j - 1];
297  text_position_axi_y = prev_option->get_y() +
298  prev_option->get_height() + TEXT_OFFSET;
299  } else {
300  /* Nothing to do. */
301  }
302 
303  Text *option = sub_options[cur_text->get_text()][j];
304  option->set_pos(text_position_axi_x, text_position_axi_y);
305  }
306  }
307 
308  // Set back button position
309  options.back()->set_pos(FONT_X, FONT_Y, true, true);
310 }
