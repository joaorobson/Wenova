95 void EditableFloor::update(float delta) {
96  InputManager *input_manager = InputManager::get_instance();
97 
98  if (input_manager->mouse_press(InputManager::LEFT_MOUSE_BUTTON)) {
99  int x = input_manager->get_mouse_x_position();
100  int y = input_manager->get_mouse_y_position();
101 
102  Rectangle mouse = Rectangle(x, y, 1, 1);
103  is_selected = Collision::is_colliding(box, mouse, rotation, 0);
104  }
105 
106  if (is_selected) {
107  static float acceleration = ACCELERATION;
108  float value = MOVE_SPEED * delta * acceleration;
109  bool moved = false;
110 
114  if (input_manager->is_key_down(InputManager::K_ARROW_RIGHT)) {
115  box.x += value;
116  moved = true;
117  }
118  if (input_manager->is_key_down(InputManager::K_ARROW_LEFT)) {
119  box.x -= value;
120  moved = true;
121  }
122  if (input_manager->is_key_down(InputManager::K_ARROW_UP)) {
123  box.y -= value;
124  moved = true;
125  }
126  if (input_manager->is_key_down(InputManager::K_ARROW_DOWN)) {
127  box.y += value;
128  moved = true;
129  }
130 
134  if (box.x < 0) {
135  box.x = 0;
136  }
137  if (box.x > BACKGROUND_WIDTH) {
138  box.x = BACKGROUND_WIDTH;
139  }
140  if (box.y < 0) {
141  box.y = 0;
142  }
143  if (box.y > BACKGROUND_HEIGHT) {
144  box.y = BACKGROUND_HEIGHT;
145  }
146 
150  if (input_manager->is_key_down(InputManager::K_ROT_LEFT)) {
151  rotation += ROTATING_SPEED * value / acceleration;
152  }
153 
157  if (input_manager->is_key_down(InputManager::K_ROT_RIGHT)) {
158  rotation -= ROTATING_SPEED * value / acceleration;
159  }
160 
164  if (input_manager->is_key_down(InputManager::K_ROT_RESET)) {
165  rotation = 0;
166  }
167 
171  if (input_manager->key_press(InputManager::K_C)) {
172  is_crossingable = not is_crossingable;
173  }
174 
178  if (input_manager->is_key_down(InputManager::K_INC_W)) {
179  standard_sprite.update_scale_x(RESIZING_SPEED * value);
180  platform_sprite.update_scale_x(RESIZING_SPEED * value);
181  selected_sprite.update_scale_x(RESIZING_SPEED * value);
182 
183  box.width = standard_sprite.get_width();
184  moved = true;
185  }
186 
190  if (input_manager->is_key_down(InputManager::K_DEC_W)) {
191  standard_sprite.update_scale_x(-RESIZING_SPEED * value);
192  platform_sprite.update_scale_x(-RESIZING_SPEED * value);
193  selected_sprite.update_scale_x(-RESIZING_SPEED * value);
194 
195  box.width = standard_sprite.get_width();
196  moved = true;
197  }
198 
199  if (moved) {
200  acceleration =
201  fmin(acceleration + ACCELERATION_INCREASE_STEP,
202  MAXIMUM_ACCELERATION);
203  } else {
204  acceleration = ACCELERATION;
205  }
206 
210  if (input_manager->is_key_down(InputManager::K_DEL)) {
211  is_deleted = true;
212  }
213  }
214 }
