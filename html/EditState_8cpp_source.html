62 void EditState::update(float delta) {

64

66  m_quit_requested = true;
67  return;
68  }
69 
70  // leave edit state
71  if (input_manager->joystick_button_press(InputManager::SELECT, 0)) {
72  music.stop();
73  sound.stop();
74  m_quit_requested = true;
75  Game::get_instance().push(new MenuState());
76  return;
77  }
78 
79  // reset position of fighter
80  if (input_manager->mouse_press(InputManager::RIGHT_MOUSE_BUTTON)) {
81  int x = input_manager->get_mouse_x_position();
82  int y = input_manager->get_mouse_y_position();
83  test_fighter->reset_position(x, y);
84  }
85 
86  // create floor or platform
87  if (input_manager->key_press(InputManager::K_F) or
88  input_manager->key_press(InputManager::K_P)) {
89  int x = input_manager->get_mouse_x_position();
90  int y = input_manager->get_mouse_y_position();
91  bool is_platform = input_manager->key_press(InputManager::K_P);
92 
93  for (auto &go : object_array) {
94  if (go->is("floor")) {
95  (static_cast<EditableFloor *>(go.get())->set_selected(false));
96  }
97  }
98 
99  auto go = new EditableFloor(x, y, 0, is_platform);
100  go->set_selected(true);
101  add_object(go);
102  }
103 
104  // save level design
105  if (input_manager->is_key_down(InputManager::K_CTRL) and
106  input_manager->key_press(InputManager::K_S)) {
107  update_level_design();
108  }
109 
110  // output
111  if (input_manager->is_key_down(InputManager::K_O)) {
112  printf("%f, %f\n", object_array[0].get()->box.x,
113  object_array[0].get()->box.y);
114  }
115 
116  if (input_manager->key_press(InputManager::K_SHIFT)) {
117  Rectangle player_box = test_fighter->box;
118  bool is_blood = test_fighter->is("blood");
119  test_fighter->kill();
120  if (is_blood)
121  test_fighter = new Flesh("test", player_box.x, player_box.y, 0);
122  else
123  test_fighter = new Blood("test", player_box.x, player_box.y, 0);
124  add_object(test_fighter);
125  }
126 
127  for (auto &background : backgrounds)
128  background.first.update(delta);
129 
130  update_array(delta);
131 }

159 void EditState::read_level_design() {
160  float x, y, width, crotation;
161  int platform;
162  ifstream level_design(RESOURCES_FOLDER + "stage_" + stage +
163  "/level_design.dat");
164  if (not level_design.is_open()) {
165  printf("Level design of stage %s can't be opened\n", stage.c_str());
166  exit(-5);
167  }
168  string s;
169  int n_backgrounds, n_sprites, speed, n_columns;
170 
171  std::getline(level_design, s);
172  for (auto &c : s)
173  c -= 15;
174  stringstream n_background_line(s);
175  n_background_line >> n_backgrounds;
176 
177  for (int i = 0; i < n_backgrounds; ++i) {
178  std::getline(level_design, s);
179  for (auto &c : s)
180  c -= 15;
181  stringstream backgrounds_line(s);
182  backgrounds_line >> x >> y >> n_sprites >> speed >> n_columns;
183  // printf("Dados: %.f %.f %d %d %d\n", x, y, n_sprites, speed,
184  // n_columns);
185  Sprite background_sprite(
186  "stage_" + stage + "/background_" + to_string(i) + ".png",
187  n_sprites, speed, n_columns);
188  Vector position(x, y);
189  backgrounds.push_back(std::make_pair(background_sprite, position));
190  }
191 
197  while (std::getline(level_design, s)) {
198  for (auto &c : s)
199  c -= 15;
200  stringstream editable_floors_line(s);
201  editable_floors_line >> x >> y >> width >> crotation >> platform;
202  // printf("Battle: %.f %.f %.f %.f\n", x, y, width, crotation);
203  add_object(new EditableFloor(x, y, width, crotation,
204  static_cast<bool>(platform)));
205  }
206  level_design.close();
207 }
