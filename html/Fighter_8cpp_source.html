152 void Fighter::notify_collision(const GameObject &object) {
153  int partner_id = (partner ? partner->get_id() : INVALID_ID);
154  int not_in_ultimate = (tags["in_ultimate"] ? 0 : 1);
155 
156  // FIXME t√° feio
157  float floor_y = object.box.y +
158  (box.x - object.box.x) * tan(object.rotation) - object.box.height * 0.5;
159 
160  if (object.is("floor") and speed.y >= 0 and
161  abs(floor_y - (box.y + box.height * 0.5)) < 10) {
162  if (pass_through_timer.get() < 30 and object.is("platform"))
163  return;
164 
165  int floor_id = ((const Floor &) object).get_id();
166  speed.y = 0;
167 
168  float new_y = object.box.y +
169  (box.x - object.box.x) * tan(object.rotation) -
170  (box.height + object.box.height) * 0.5;
171 
172  if (last_collided_floor != floor_id)
173  box.y = min(box.y, new_y);
174  else
175  box.y = new_y;
176 
177  on_floor = true;
178  last_collided_floor = ((const Floor &) object).get_id();
179  } else if (object.is("player") and not is("dying") and
180  not(object.is("dying"))) {
181  Fighter const &fighter = (const Fighter &) object;
182 
183  if (fighter.is_attacking() and fighter.get_id() != partner_id) {
184  int left = AttackDirection::ATK_LEFT * (fighter.box.x > box.x);
185  int right = AttackDirection::ATK_RIGHT * (fighter.box.x <= box.x);
186  int up = AttackDirection::ATK_UP * (fighter.box.y > box.y);
187  int down = AttackDirection::ATK_DOWN * (fighter.box.y <= box.y);
188  int position_mask = left | right | up | down;
189 
190  if (position_mask & fighter.get_attack_mask()) {
191  float damage = fighter.get_attack_damage() *
192  ((state == FighterState::DEFENDING) ? 0.5 : 1);
193  increment_life(-damage);
194  float m_increment_special = (fighter.get_attack_damage() / 3) *
195  ((state == FighterState::DEFENDING) ? 0 : 1) *
196  not_in_ultimate;
197  this->increment_special(m_increment_special);
198  if (state != FighterState::DEFENDING)
199  check_stunned();
200  }
201  } else if (is_attacking() and fighter.get_id() != partner_id) {
202  int left = AttackDirection::ATK_LEFT * (fighter.box.x <= box.x);
203  int right = AttackDirection::ATK_RIGHT * (fighter.box.x > box.x);
204  int up = AttackDirection::ATK_UP * (fighter.box.y <= box.y);
205  int down = AttackDirection::ATK_DOWN * (fighter.box.y > box.y);
206  int position_mask = left | right | up | down;
207 
208  if (position_mask & get_attack_mask()) {
209  grab = true;
210  play_hit();
211  this->increment_special((attack_damage / 2) * not_in_ultimate);
212  }
213  }
214  }
215 
216  change_state(temporary_state);
217 }
