186 void Sprite::update(float delta_time) {
187  time_elapsed += delta_time;
193  if (time_elapsed >= frame_time) {
194  time_elapsed = 0;
195 
196  current_frame = current_frame + 1;
200  if (current_frame == frame_count) {
201  finished = true;
202  current_frame = 0;
203  }
204  set_clip((current_frame % columns) * width,
205  (current_frame / columns) * height,
206  width,
207  height);
208  }
