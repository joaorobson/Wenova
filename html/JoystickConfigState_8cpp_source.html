249 void JoystickConfigState::update(float delta) {
250  InputManager *input_manager = InputManager::get_instance();
251  assert(input_manager != nullptr);
252 
253  assert(delta >= 0);
254 
258  if (input_manager->quit_requested()) {
259  m_quit_requested = true;
260  return;
261  } else {
262  /* Nothing to do. */
263  }
264 
268  if (not on_test) {
272  if (input_manager->joystick_button_press(InputManager::SELECT, 0) or
273  input_manager->joystick_button_press(InputManager::B, 0)) {
274  selected.play();
275  m_quit_requested = true;
276  Game::get_instance().push(new OptionsState());
277  return;
278  } else {
282  if (input_manager->joystick_button_press(InputManager::A, 0) and
283  not is_keyboard) {
284  selected.play();
285  on_test = true;
286  } else {
287  /* Nothing to do. */
288  }
289  }
290  } else {
294  if (input_manager->is_joystick_button_down(InputManager::SELECT, 0) and
295  input_manager->is_joystick_button_down(InputManager::START, 0)) {
296  selected.play();
297  on_test = false;
298  InputManager::get_instance()->map_keyboard_to_joystick(
299  InputManager::MENU_MODE);
300  } else {
304  if (input_manager->joystick_button_release(InputManager::A, 0)) {
305  InputManager::get_instance()->map_keyboard_to_joystick(
306  InputManager::BATTLE_MODE);
307  } else {
308  /* Nothing to do. */
309  }
310  }
311  }
312 
313  update_array(delta);
314 }
