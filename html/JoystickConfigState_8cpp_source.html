173 void JoystickConfigState::update(float delta) {
174  InputManager *input_manager = InputManager::get_instance();
175 
179  if (input_manager->quit_requested()) {
180  m_quit_requested = true;
181  return;
182  }
183 
187  if (on_test) {
191  if (input_manager->is_joystick_button_down(InputManager::SELECT, 0) and
192  input_manager->is_joystick_button_down(InputManager::START, 0)
193  ) {
194  selected.play();
195  on_test = false;
196  InputManager::get_instance()->map_keyboard_to_joystick(
197  InputManager::MENU_MODE);
198  }
199 
203  if (input_manager->joystick_button_release(InputManager::A, 0)) {
204  InputManager::get_instance()->map_keyboard_to_joystick(
205  InputManager::BATTLE_MODE);
206  }
207  } else {
211  if (input_manager->joystick_button_press(InputManager::SELECT, 0) or
212  input_manager->joystick_button_press(InputManager::B, 0)) {
213  selected.play();
214  m_quit_requested = true;
215  Game::get_instance().push(new OptionsState());
216  return;
217  }
218 
222  if (input_manager->joystick_button_press(InputManager::A, 0) and
223  not is_keyboard
224  ) {
225  selected.play();
226  on_test = true;
227  }
228  }
229 
230  update_array(delta);
231 }
