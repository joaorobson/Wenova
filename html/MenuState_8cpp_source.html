89 void MenuState::update(float delta) {
90  title.update(delta);
91  planet.update(delta);
92  green_ship.update(delta);
93  red_ship.update(delta);
94 
95  process_input();
96 
97  InputManager* input_manager = InputManager::get_instance();
98 
99  if (input_manager->quit_requested() or pressed[SELECT] or pressed[B]) {
100  music.stop();
101  m_quit_requested = true;
102  return;
103  }
104 
105  // handling options input
106  if (pressed[LEFT] and current_option != 0) {
107  changed.play();
108  current_option--;
109  }
110 
111  if (pressed[RIGHT] and
112  current_option != static_cast<int>(options.size()) - 1) {
113  changed.play();
114  current_option++;
115  }
116 
117  if (is_holding[LB] and is_holding[RT] and is_holding[Y]) {
118  m_quit_requested = true;
119  Game::get_instance().push(new EditState("2"));
120  return;
121  }
122 
123  if (pressed[START] or pressed[A]) {
124  selected.play();
125 
126  if (not start_pressed) {
127  start_pressed = true;
128  current_option = 0;
129  } else {
130  m_quit_requested = true;
131 
132  if (current_option == 0) {
133  Game::get_instance().push(new StageSelectState());
134  } else if (current_option == 1) {
135  Game::get_instance().push(new StageSelectState(true));
136  } else if (current_option == 2) {
137  Game::get_instance().push(new OptionsState());
138  }
139 
140  return;
141  }
142  }
143 
144  if (is_holding[LB] and is_holding[RT] and is_holding[Y]) {
145  m_quit_requested = true;
146  Game::get_instance().push(new EditState("1"));
147  return;
148  }
149 
150  if (start_pressed) {
151  // handling options positioning
152  options[current_option]->set_pos(FONT_X, FONT_Y, true, true);
153  options[current_option]->set_color(LIGHT_GREEN);
154 
155  // positioning options before current option
156  for (int idx = 0; idx < current_option; idx++) {
157  Text* next_option = options[idx + 1];
158 
159  int new_x = next_option->get_x() - options[idx]->get_width() -
160  OPTION_OFFSET;
161  options[idx]->set_pos(new_x, FONT_Y, false, true);
162  options[idx]->set_color(WHITE);
163  }
164 
165  // positioning options after current option
166  for (unsigned int idx = current_option + 1; idx < options.size();
167  idx++) {
168  Text* prev_option = options[idx - 1];
169 
170  int new_x =
171  prev_option->get_x() + prev_option->get_width() + OPTION_OFFSET;
172  options[idx]->set_pos(new_x, FONT_Y, false, true);
173  options[idx]->set_color(WHITE);
174  }
175  }
176 
177  if (text_timer.get() > TEXT_TIMER_COOLDOWN) {
178  show_text = !show_text;
179  text_timer.restart();
180  }
181 
182  text_timer.update(delta);
183 }
