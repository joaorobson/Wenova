197 void CharacterSelectState::update(float delta) {

214  (not is_character_selected[FIRST_PLAYER] and
215  is_key_pressed[FIRST_PLAYER][B])) {
216  select_sound.play();
217  m_quit_requested = true;
218  Game::get_instance().push(new StageSelectState());
219  return;
220  }
221 
225  if (all_players_selected()) {
226  is_ready = true;
227  if (is_key_pressed[FIRST_PLAYER][START] or
228  is_key_pressed[FIRST_PLAYER][A]) {
229  select_sound.play();

235  }
236  }
237 
241  for (int i = 0; i < N_PLAYERS; i++) {
242  if (not is_character_selected[i]) {
246  if (is_key_pressed[i][Y]) {
247  select_sound.play();
248  int rand_col = 0, rand_row = 0, rand_skin = 0, char_sel = 0;
249 
253  do {
254  unsigned int seed1 = clock();
255  unsigned int seed2 = clock();
256 
257  rand_col = rand_r(&seed1) % N_COLS;
258  rand_row = rand_r(&seed2) % N_ROWS;
259 
260  char_sel = rand_row * N_COLS + rand_col;
261  } while (not chars[char_sel].is_enabled());
262 
266  do {
267  unsigned int seed = clock();
268 
269  rand_skin = rand_r(&seed) % N_SKINS;
270  } while (not chars[char_sel].is_skin_available(rand_skin));
271 
272  current_column[i] = rand_col;
273  current_row[i] = rand_row;
274  current_skin[i] = rand_skin;
275  }
276 
277  int old_col = current_column[i];
278  int old_row = current_row[i];
279 
283  if (is_key_pressed[i][LEFT]) {
284  changed_sound.play();
285  if (current_column[i] != 0) {
286  current_column[i]--;
287  }
288  }
289  if (is_key_pressed[i][RIGHT]) {
290  changed_sound.play();
291  if (current_column[i] + 1 < N_COLS) {
292  current_column[i]++;
293  }
294  }
295  if (is_key_pressed[i][UP]) {
296  changed_sound.play();
297  if (current_row[i] != 0) {
298  current_row[i]--;
299  }
300  }
301  if (is_key_pressed[i][DOWN]) {
302  changed_sound.play();
303  if (current_row[i] + 1 < N_ROWS) {
304  current_row[i]++;
305  }
306  }
307 
311  if (current_column[i] != old_col or current_row[i] != old_row) {
312  current_skin[i] = 0;
313  }
314 
318  if (is_key_pressed[i][LT]) {
319  changed_sound.play();
320  current_skin[i] = (current_skin[i] - 1 + N_SKINS) % N_SKINS;
321  }
322  if (is_key_pressed[i][RT]) {
323  changed_sound.play();
324  current_skin[i] = (current_skin[i] + 1) % N_SKINS;
325  }
326 
330  if (is_key_pressed[i][A]) {
331  int char_sel = current_row[i] * N_COLS + current_column[i];
332 
336  if (chars[char_sel].is_enabled()) {
337  if (not chars[char_sel].is_skin_available(
338  current_skin[i])) {
339  blocked_sound.play();
340  } else {
341  select_sound.play();
342  chars[char_sel].lock_skin(current_skin[i]);
343  is_character_selected[i] = true;
344  }
345  } else {
346  blocked_sound.play();
347  }
348  }
349  } else {
353  if (is_key_pressed[i][B]) {
354  int char_sel = current_row[i] * N_COLS + current_column[i];
355  chars[char_sel].unlock_skin(current_skin[i]);
356  is_character_selected[i] = false;
357  is_ready = false;
358  }
359  }
360  }
361 
365  for (int i = 0; i < N_CHARS; i++) {
366  chars[i].get_disabled().update(delta);
367 
368  for (int j = 0; j < N_SKINS; j++) {
369  chars[i].get_skin(j).update(delta);
370  }
371  }
372 
373  planet_sprite.update(delta);
374 }
