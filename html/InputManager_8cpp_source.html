154 void InputManager::update() {
155  SDL_Event event;
156 
157  has_quit_request = false;
158 
159  update_counter++;
160 
161  SDL_GetMouseState(&mouse_x_position, &mouse_y_position);
162 
163  mouse_x_position = mouse_x_position * mouse_sensibility_value + offset_x;
164  mouse_y_position = mouse_y_position * mouse_sensibility_value + offset_y;
165 
166  mouse_x_position = std::max(0, mouse_x_position);
167  mouse_x_position = std::min(mouse_x_position, BACKGROUND_WIDTH);
168  mouse_y_position = std::max(0, mouse_y_position);
169  mouse_y_position = std::min(mouse_y_position, BACKGROUND_HEIGHT);
170 
175  while (SDL_PollEvent(&event)) {
176  int key_id;
177  int button_id;
178  int joystick_id = controllers_id[event.cdevice.which];
179 
186  switch (event.type) {
187  case SDL_KEYDOWN:
188  if (event.key.repeat) {
189  break;
190  }
191  key_id = event.key.keysym.sym;
192  keys_states[key_id] = true;
193  keys_updates[key_id] = update_counter;
194  emulate_joystick(key_id, true);
195  break;
196 
197  case SDL_KEYUP:
198  key_id = event.key.keysym.sym;
199  keys_states[key_id] = false;
200  keys_updates[key_id] = update_counter;
201  emulate_joystick(key_id, false);
202  break;
203 
204  case SDL_MOUSEBUTTONDOWN:
205  button_id = event.button.button;
206  mouse_buttons_states[button_id] = true;
207  mouse_update[button_id] = update_counter;
208  break;
209 
210  case SDL_MOUSEBUTTONUP:
211  button_id = event.button.button;
212  mouse_buttons_states[button_id] = false;
213  mouse_update[button_id] = update_counter;
214  break;
215 
216  case SDL_JOYAXISMOTION:
217  break;
218 
219  case SDL_CONTROLLERAXISMOTION:
220  if (event.caxis.axis == SDL_CONTROLLER_AXIS_LEFTX) {
221  joysticks_buttons_states[joystick_id][RIGHT] =
222  event.caxis.value > analogic_sensibility_value;
223  joysticks_buttons_states[joystick_id][LEFT] =
224  event.caxis.value < -analogic_sensibility_value;
225  joystick_update[joystick_id][RIGHT] = update_counter;
226  joystick_update[joystick_id][LEFT] = update_counter;
227  } else if (event.caxis.axis == SDL_CONTROLLER_AXIS_LEFTY) {
228  joysticks_buttons_states[joystick_id][DOWN] =
229  event.caxis.value > analogic_sensibility_value;
230  joysticks_buttons_states[joystick_id][UP] =
231  event.caxis.value < -analogic_sensibility_value;
232  joystick_update[joystick_id][DOWN] = update_counter;
233  joystick_update[joystick_id][UP] = update_counter;
234  } else if (event.caxis.axis ==
235  SDL_CONTROLLER_AXIS_TRIGGERLEFT) {
236  joysticks_buttons_states[joystick_id][LT] =
237  event.caxis.value > triggers_sensibility_value;
238  joystick_update[joystick_id][LT] = update_counter;
239  } else if (event.caxis.axis ==
240  SDL_CONTROLLER_AXIS_TRIGGERRIGHT) {
241  joysticks_buttons_states[joystick_id][RT] =
242  event.caxis.value > triggers_sensibility_value;
243  joystick_update[joystick_id][RT] = update_counter;
244  }
245 
246  break;
247 
248  case SDL_JOYBUTTONDOWN:
249  break;
250 
251  case SDL_CONTROLLERBUTTONDOWN:
252  button_id = event.cbutton.button;
253  joysticks_buttons_states[joystick_id][button_id] = true;
254  joystick_update[joystick_id][button_id] = update_counter;
255  break;
256 
257  case SDL_JOYBUTTONUP:
258  break;
259 
260  case SDL_CONTROLLERDEVICEADDED:
261  connect_joysticks();
262  break;
263 
264  case SDL_CONTROLLERDEVICEREMOVED:
265  connect_joysticks();
266  break;
267 
268  case SDL_CONTROLLERBUTTONUP:
269  button_id = event.cbutton.button;
270  joysticks_buttons_states[joystick_id][button_id] = false;
271  joystick_update[joystick_id][button_id] = update_counter;
272  break;
273 
274  case SDL_QUIT:
275  has_quit_request = true;
276  break;
277 
278  default:
279  break;
280  }
281  }

457 void InputManager::connect_joysticks() {
461  int max = SDL_NumJoysticks();
462  if (max > N_CONTROLLERS) {
463  max = N_CONTROLLERS;
464  }
465  int n_controller = 0;
466 
470  for (int i = 0; i < max; i++) {
471  if (controllers[i] != nullptr) {
472  SDL_GameControllerClose(controllers[i]);
473  controllers[i] = nullptr;
474  }
475  }
476 
480  for (int i = 0; i < max; i++) {
481  char guid[64];
482  SDL_JoystickGetGUIDString(SDL_JoystickGetDeviceGUID(i), guid,
483  sizeof(guid));
484 
485  if (SDL_IsGameController(i)) {
486  controllers[i] = SDL_GameControllerOpen(i);
487 
488  SDL_Joystick *j = SDL_GameControllerGetJoystick(controllers[i]);
489  int instance_id = SDL_JoystickInstanceID(j);
490  printf("Controller %d (%d real) connected\n", i, instance_id);
491  controllers_id[instance_id] = i;
492  n_controller++;
493  } else {
494  printf("WARNING: Joystick is not a game controller\n");
495  SDL_JoystickOpen(i);
496  }
497  }
498 }

533 void InputManager::emulate_joystick(int key_id, bool state) {
534  if (state) {
538  switch (key_id) {
539  case SDLK_0:
540  reset_keyboard_to_joystick();
541  keyboard_to_joystick_id = -1;
542  break;
543  case SDLK_1:
544  reset_keyboard_to_joystick();
545  keyboard_to_joystick_id = 0;
546  break;
547  case SDLK_2:
548  reset_keyboard_to_joystick();
549  keyboard_to_joystick_id = 1;
550  break;
551  case SDLK_3:
552  reset_keyboard_to_joystick();
553  keyboard_to_joystick_id = 2;
554  break;
555  case SDLK_4:
556  reset_keyboard_to_joystick();
557  keyboard_to_joystick_id = 3;
558  break;
559  case SDLK_5:
560  reset_keyboard_to_joystick();
561  keyboard_to_joystick_id = 4;
562  break;
563  }
564  }
565 
569  if (keyboard_to_joystick_id == N_CONTROLLERS) {
570  for (int i = 0; i < N_CONTROLLERS; i++) {
571  joysticks_buttons_states[i][keyboard_to_joystick[key_id] - 1] =
572  state;
573  joystick_update[i][keyboard_to_joystick[key_id] - 1] =
574  update_counter;
575  }
576  } else if (keyboard_to_joystick_id >= 0) {
577  joysticks_buttons_states[keyboard_to_joystick_id]
578  [keyboard_to_joystick[key_id] - 1] = state;
579  joystick_update[keyboard_to_joystick_id]
580  [keyboard_to_joystick[key_id] - 1] = update_counter;
581  }
582 }
