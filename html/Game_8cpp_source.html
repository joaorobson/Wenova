232 void Game::update_resolution() {

241 
242  float original_ratio = 1280.0 / 720;
243  float new_ratio = (1.0 * width) / height;
244  float offset_x, offset_y, w, h;
245 
246  if (original_ratio < new_ratio) {
247  h = height;
248  w = height * original_ratio;
249  offset_y = 0;
250  offset_x = (720 / h) * (w - width) / 2;
251  } else {
252  w = width;
253  h = width / original_ratio;
254  offset_x = 0;
255  offset_y = (1280 / w) * (h - height) / 2;
256  }
257 
258  InputManager::get_instance()->set_mouse_sensibility_value(
259  1280 / w, offset_x, offset_y);
